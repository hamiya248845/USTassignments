let form = document.createElement("form"); let textBox = document.createElement("input"); textBox.type = "text"; textBox.placeholder = "Enter a number between 0 and 6"; form.appendChild(textBox); // Create a button let button = document.createElement("button"); button.textContent = "Submit"; form.appendChild(button); // Add form to the document document.body.appendChild(form); // Function to handle form submission function handleSubmit(event) { event.preventDefault(); // Get the input value let inputValue = parseInt(textBox.value); // Check if the input value is within the range of 0 to 6 if (inputValue >= 0 && inputValue <= 6) { // Handle the input value switch (inputValue) { case 0: console.log("Sunday"); break; case 1: console.log("Monday"); break; case 2: console.log("Tuesday"); break; case 3: console.log("Wednesday"); break; case 4: console.log("Thursday"); break; case 5: console.log("Friday"); break; case 6: console.log("Saturday"); break; } } else { // Throw an exception for invalid input throw new Error("Invalid input! Please enter a number between 0 and 6."); } } // Attach event listener to the form submission form.addEventListener("submit", handleSubmit);
